#!/usr/bin/env python3
"""
Consolidate Capability Investment Case - Final Version
Uses exact checkbox and label columns from your mapping
"""

import argparse
from pathlib import Path
from datetime import datetime
import zipfile
from openpyxl import load_workbook, Workbook


def is_valid_xlsx(file_path):
    try:
        return zipfile.is_zipfile(file_path)
    except:
        return False


def extract_from_sheet(wb):
    ws = wb["1_Financial Investment Case"] if "1_Financial Investment Case" in wb.sheetnames else wb.active

    def get_cell(col, row):
        try:
            val = ws.cell(row=row, column=col).value
            return str(val).strip() if val is not None else ""
        except:
            return ""

    data = {}

    # === Basic Fields ===
    data["Capability_Name"] = get_cell(4, 6)
    data["Funder"] = get_cell(4, 7)
    data["Builder_SVS"] = get_cell(4, 8)
    data["Funding_Type"] = get_cell(4, 9)
    data["Investment_Themes"] = get_cell(4, 10)
    data["Discovery_Maturity"] = get_cell(4, 11)
    data["Business_Capability"] = get_cell(4, 12)
    data["JIRA_ID"] = get_cell(4, 13)

    # === FSA Category (G12:G15 checkbox, F12:F15 label) ===
    fsa = []
    for r in range(12, 16):
        if get_cell(7, r) in ["☑", "TRUE", "x", "X", "1", True]:
            fsa.append(get_cell(6, r))          # Label from Column F
    data["FSA_Category"] = ", ".join([x for x in fsa if x])

    # === Financial ===
    data["T_Shirt_Size"] = get_cell(11, 6)
    data["Tech_Investment_2027_k"] = get_cell(11, 7)
    data["Multi_Year_Funding"] = get_cell(11, 10)

    # === Customer Segment (G6:G9 checkbox, F6:F9 label) ===
    segments = []
    for r in range(6, 10):
        if get_cell(7, r) in ["☑", "TRUE", "x", "X", "1", True]:
            segments.append(get_cell(6, r))     # Label from Column F
    data["Customer_Segment"] = ", ".join([x for x in segments if x])

    # === Markets Impacted (G18:G39 + H18:H39 checkbox, F18:F39 label) ===
    markets = []
    for r in range(18, 40):
        if get_cell(7, r) in ["☑", "TRUE", "x", "X", "1", True] or get_cell(8, r) in ["☑", "TRUE", "x", "X", "1", True]:
            markets.append(get_cell(6, r))      # Label from Column F
    data["Markets_Impacted"] = ", ".join([x for x in markets if x])

    # === Other Fields ===
    data["Business_Outcome_Lead"] = get_cell(4, 16)
    data["Capability_Description"] = get_cell(3, 20)
    data["Investment_Case_Breakdown"] = get_cell(3, 42)[:1000]
    data["Business_Benefit_WHY"] = get_cell(10, 26)
    data["OKR_Summary"] = get_cell(10, 42)

    return data


def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("folder", nargs="?", default=".", help="Root folder to scan")
    parser.add_argument("-o", "--output", default="Consolidated_Capabilities.xlsx")
    args = parser.parse_args()

    root = Path(args.folder).resolve()
    output_file = Path(args.output)

    print(f"🔍 Scanning: {root}\n")

    records = []

    for file_path in sorted(root.rglob("*.xlsx")):
        if "~$" in file_path.name:
            continue
        if not is_valid_xlsx(file_path):
            print(f"⚠️ Skipped (invalid xlsx): {file_path.name}")
            continue

        try:
            wb = load_workbook(file_path, data_only=True, read_only=True)
            row_data = extract_from_sheet(wb)

            row_data["File_Name"] = file_path.name
            row_data["Folder"] = str(file_path.parent.relative_to(root))
            row_data["Date_Read"] = datetime.now().strftime("%Y-%m-%d %H:%M")

            records.append(row_data)
            print(f"✓ {file_path.name}")

        except Exception as e:
            print(f"✗ Failed {file_path.name}: {e}")

    if not records:
        print("No files processed.")
        return

    # Write output
    wb_out = Workbook()
    ws = wb_out.active
    ws.title = "Consolidated"

    headers = ["File_Name", "Folder", "Date_Read", "Capability_Name", "Funder", "Builder_SVS",
               "Funding_Type", "Investment_Themes", "Discovery_Maturity", "Business_Capability",
               "JIRA_ID", "FSA_Category", "T_Shirt_Size", "Tech_Investment_2027_k",
               "Multi_Year_Funding", "Customer_Segment", "Markets_Impacted",
               "Business_Outcome_Lead", "Capability_Description", "Investment_Case_Breakdown",
               "Business_Benefit_WHY", "OKR_Summary"]

    for col, h in enumerate(headers, 1):
        ws.cell(1, col, h)

    for r_idx, row in enumerate(records, 2):
        for c_idx, h in enumerate(headers, 1):
            ws.cell(r_idx, c_idx, row.get(h, ""))

    wb_out.save(output_file)
    print(f"\n✅ Saved {len(records)} rows to {output_file}")


if __name__ == "__main__":
    main()
